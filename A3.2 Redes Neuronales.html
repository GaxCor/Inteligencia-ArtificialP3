import numpy as np
import matplotlib.pyplot as plt
from tensorflow.keras.datasets import mnist
import random
import cv2

# 1. Cargar MNIST
(train_images, train_labels), (_, _) = mnist.load_data()

# 2. Seleccionar 5 imgenes por dgito
digit_images = {i: [] for i in range(10)}
for img, label in zip(train_images, train_labels):
    if len(digit_images[label]) < 5:
        digit_images[label].append(img)
    if all(len(lst) == 5 for lst in digit_images.values()):
        break

# 3. Funcin para modificar imagen (pequeas rotaciones y ruidos)
def modify_image(img):
    angle = random.randint(-20, 20)
    M = cv2.getRotationMatrix2D((14, 14), angle, 1.0)
    rotated = cv2.warpAffine(img, M, (28, 28), borderMode=cv2.BORDER_CONSTANT, borderValue=0)
    
    # Agregar ruido aleatorio
    noise = np.random.normal(0, 10, (28, 28))
    noisy = rotated + noise
    noisy = np.clip(noisy, 0, 255)
    return noisy.astype(np.uint8)

# 4. Generar conjunto de 50 imgenes modificadas
generated_images = []
generated_labels = []

for digit, imgs in digit_images.items():
    for img in imgs:
        modified = modify_image(img)
        generated_images.append(modified)
        generated_labels.append(digit)

# 5. Mostrar ejemplos
fig, axes = plt.subplots(5, 10, figsize=(15, 8))
for i, ax in enumerate(axes.flat):
    ax.imshow(generated_images[i], cmap='gray')
    ax.set_title(f"{generated_labels[i]}")
    ax.axis('off')
plt.tight_layout()
plt.show()

Evaluacin del desempeo en las imgenes generadas

Despus de generar las 50 imgenes modificadas (mediante rotaciones y ruido), se evalu el desempeo del modelo.

Al tratarse de imgenes ms distorsionadas respecto a los datos originales del dataset MNIST, es esperable que el modelo cometa ms errores.

Motivos de potencial error:

  • Rotaciones: El modelo puede ser sensible a la inclinacin del nmero, ya que fue entrenado con dgitos orientados verticalmente.
  • Ruido: La adicin de ruido aleatorio altera los pxeles importantes para la prediccin, dificultando que la red encuentre las caractersticas aprendidas.
  • Variaciones no vistas: El modelo fue entrenado en un conjunto relativamente limpio (MNIST), por lo que no necesariamente generaliza bien a deformaciones no presentes en el entrenamiento.

Comparacin con desempeo previo:

  • En el conjunto original de MNIST, el modelo alcanz una precisin muy alta (>98%).
  • En las imgenes modificadas, se observ una disminucin en la precisin, evidenciando que la robustez ante variaciones es limitada sin tcnicas de aumento de datos durante el entrenamiento.

Esta comparacin destaca la importancia de tcnicas como data augmentation o entrenamientos con datasets ms diversos para mejorar la capacidad de generalizacin de las redes neuronales.

import cv2
import numpy as np
import tensorflow as tf

# Cargar el modelo
model = tf.keras.models.load_model("my_model.h5")

# Iniciar cmara
cap = cv2.VideoCapture(0)

try:
    while True:
        ret, frame = cap.read()
        if not ret:
            break

        # Convertir a escala de grises
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

        # Aplicar umbral binario inverso
        _, threshed = cv2.threshold(gray, 100, 255, cv2.THRESH_BINARY_INV)

        # Redimensionar a 28x28
        resized = cv2.resize(threshed, (28, 28))

        # Preparar imagen para la red neuronal
        input_img = resized.reshape(1, 28, 28, 1) / 255.0

        # Prediccin
        prediction = model.predict(input_img)
        predicted_label = np.argmax(prediction)

        # Mostrar prediccin sobre el frame
        frame_with_prediction = cv2.putText(
            frame.copy(), str(predicted_label), (10, 30),
            cv2.FONT_HERSHEY_SIMPLEX, 1, (255, 0, 0), 2, cv2.LINE_AA
        )

        # Mostrar ventanas
        cv2.imshow('Real-time prediction', frame_with_prediction)

        # Mostrar imagen procesada (escalada para visualizacin)
        upscaled_resized = cv2.resize(resized, (280, 280), interpolation=cv2.INTER_NEAREST)
        cv2.imshow('Resized Image', upscaled_resized)

        # Salir con 'q'
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

except KeyboardInterrupt:
    pass
finally:
    cap.release()
    cv2.destroyAllWindows()

6. Mejoras realizadas al sistema

N Mejora Descripcin Beneficio
1 Definicin de ROI (Regin de Inters) Se recorta un rea de 200x200 pxeles en el centro de la imagen para realizar la prediccin. Mejora el enfoque en el nmero y evita errores causados por elementos del fondo.
2 Guas visuales (lneas de cruz) Se dibujan lneas horizontales y verticales cruzadas en el centro de la pantalla. Ayuda al usuario a alinear correctamente el nmero dentro del rea de inters.
3 Cuadro delimitador azul Se dibuja un rectngulo azul alrededor del rea ROI donde debe colocarse el nmero. Da una referencia visual precisa de dnde colocar el nmero para mejor prediccin.
4 Visualizacin de porcentaje de confianza Se muestra junto a la prediccin el nivel de confianza del modelo en el nmero detectado. Permite interpretar la seguridad de la prediccin y saber cundo confiar o dudar.
5 Opcional: Desenfoque ligero Se incluye la opcin de aplicar GaussianBlur para reducir el ruido en la imagen antes de umbralizar (comentado en el cdigo). Puede mejorar la prediccin con numeros bien marcados, pero numeros "cortados" o no muy visibles, empeora el modelo.

import cv2
import numpy as np
import tensorflow as tf

# Cargar el modelo
model = tf.keras.models.load_model("my_model.h5")

# Iniciar cmara
cap = cv2.VideoCapture(0)

# Variables para botn "Salir"
button_pressed = False
button_x1, button_y1, button_x2, button_y2 = 10, 10, 110, 50

# Funcin para detectar clics
def click_event(event, x, y, flags, param):
    global button_pressed
    if event == cv2.EVENT_LBUTTONDOWN:
        if button_x1 <= x <= button_x2 and button_y1 <= y <= button_y2:
            button_pressed = True

# Asignar funcin de clic
cv2.namedWindow('Real-time prediction')
cv2.setMouseCallback('Real-time prediction', click_event)

try:
    while True:
        ret, frame = cap.read()
        if not ret:
            break

        height, width = frame.shape[:2]

        # Definir ROI centrado
        roi_size = 200
        x1 = width // 2 - roi_size // 2
        y1 = height // 2 - roi_size // 2
        x2 = x1 + roi_size
        y2 = y1 + roi_size

        # Dibujar guas y cuadro de referencia
        guide_color = (0, 255, 0)
        thickness = 1
        cv2.line(frame, (width // 2, 0), (width // 2, height), guide_color, thickness)
        cv2.line(frame, (0, height // 2), (width, height // 2), guide_color, thickness)
        cv2.rectangle(frame, (x1, y1), (x2, y2), (255, 0, 0), 2)

        # Extraer y procesar el ROI
        roi = frame[y1:y2, x1:x2]
        gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)

        # (Opcional) Desenfoque ligero para reducir ruido
        # gray = cv2.GaussianBlur(gray, (3, 3), 0)
